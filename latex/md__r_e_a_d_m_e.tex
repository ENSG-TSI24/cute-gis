\begin{DoxyCode}{0}

\end{DoxyCode}

\item Pour le back-\/end 
\begin{DoxyCode}{0}
\DoxyCodeLine{git checkout -\/b <devback/nom\_de\_fonction>}

\end{DoxyCode}
